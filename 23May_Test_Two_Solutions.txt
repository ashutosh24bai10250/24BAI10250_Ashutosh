Q1.  List the Employee Name, their Department Name, and the Location Name for all employees.
     SELECT E.employee_name, D.dept_name, D.location_id
     FROM Employees E
     JOIN Departments D ON E.dept_id = D.dept_id;

Q2.  Retrieve the Project Name, Client Name, and the Total Hours Worked on that project,
     but only for projects managed by the 'Consulting' department.
     SELECT P.project_name, P.client_name, SUM(A.hours_worked) AS total_hours
     FROM Projects P
     JOIN Departments D ON P.dept_id = D.dept_id
     JOIN Assignments A ON P.project_id = A.project_id
     WHERE D.dept_name = 'Consulting'
     GROUP BY P.project_name, P.client_name;

Q3.  Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'.
     List the Employee Name and Project Name.
     SELECT E.employee_name, P.project_name
     FROM Employees E
     JOIN Assignments A ON E.employee_id = A.employee_id
     JOIN Projects P ON A.project_id = P.project_id
     WHERE P.client_name = 'GlobalTech Solutions';

Q4.  List all Departments that have no Employees currently assigned to them.
     SELECT D.dept_name
     FROM Departments D
     LEFT JOIN Employees E ON D.dept_id = E.dept_id
     WHERE E.employee_id IS NULL;

Q5.  Show the Manager Name, the Department Name they manage, and the total number of direct reports they have.
     SELECT M.employee_name AS Manager_Name, D.dept_name, COUNT(E.employee_id) AS direct_reports
     FROM Employees E
     JOIN Employees M ON E.manager_id = M.employee_id
     JOIN Departments D ON M.dept_id = D.dept_id
     GROUP BY M.employee_name, D.dept_name;

Q6.  Calculate the Average Hours Worked per Employee across all projects.
     List the Employee Name and the average hours.
     SELECT E.employee_name, AVG(A.hours_worked) AS avg_hours
     FROM Employees E
     JOIN Assignments A ON E.employee_id = A.employee_id
     GROUP BY E.employee_name;

Q7.  Find the Department Name(s) that have an average salary above the company-wide average salary.
     SELECT D.dept_name, AVG(E.salary) AS avg_salary
     FROM Employees E
     JOIN Departments D ON E.dept_id = D.dept_id
     GROUP BY D.dept_name
     HAVING AVG(E.salary) > (SELECT AVG(salary) FROM Employees);

Q8.  Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.
     SELECT P.client_name, SUM(A.hours_worked) AS total_hours
     FROM Projects P
     JOIN Assignments A ON P.project_id = A.project_id
     GROUP BY P.client_name
     HAVING SUM(A.hours_worked) > 500;

Q9.  For each Location, find the Department Name with the highest count of employees.
     SELECT location_id, dept_name, emp_count
     FROM (
         SELECT D.location_id, D.dept_name, COUNT(E.employee_id) AS emp_count,
                RANK() OVER (PARTITION BY D.location_id ORDER BY COUNT(E.employee_id) DESC) AS rnk
         FROM Departments D
         JOIN Employees E ON D.dept_id = E.dept_id
         GROUP BY D.location_id, D.dept_name
     ) ranked
     WHERE rnk = 1;

Q10. Calculate the total number of employees assigned to each Project,
     but only include projects that have more than 5 employees assigned.
     SELECT P.project_name, COUNT(A.employee_id) AS total_employees
     FROM Projects P
     JOIN Assignments A ON P.project_id = A.project_id
     GROUP BY P.project_name
     HAVING COUNT(A.employee_id) > 5;

Q11. Find the Employee Name(s) who work on ALL projects managed by the 'R&D' department.
     SELECT E.employee_name
     FROM Employees E
     WHERE NOT EXISTS (
         SELECT P.project_id
         FROM Projects P
         JOIN Departments D ON P.dept_id = D.dept_id
         WHERE D.dept_name = 'R&D'
         AND P.project_id NOT IN (
             SELECT A.project_id
             FROM Assignments A
             WHERE A.employee_id = E.employee_id
         )
     );

Q12. Retrieve the Department Name(s) that have employees assigned to more than one project.
     SELECT DISTINCT D.dept_name
     FROM Departments D
     JOIN Employees E ON D.dept_id = E.dept_id
     JOIN Assignments A ON E.employee_id = A.employee_id
     GROUP BY D.dept_name, E.employee_id
     HAVING COUNT(DISTINCT A.project_id) > 1;

Q13. List the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.
     SELECT E.employee_name, P.project_name, A.hours_worked
     FROM Employees E
     JOIN Assignments A ON E.employee_id = A.employee_id
     JOIN Projects P ON A.project_id = P.project_id
     WHERE A.hours_worked < 10;

Q14. Find the Project Name(s) that have no assigned employees.
     SELECT P.project_name
     FROM Projects P
     LEFT JOIN Assignments A ON P.project_id = A.project_id
     WHERE A.assignment_id IS NULL;

Q15. Select the Employee Name(s) whose salary is greater than the maximum salary in their own department.
     SELECT E.employee_name, E.salary
     FROM Employees E
     WHERE E.salary > (
         SELECT MAX(E2.salary)
         FROM Employees E2
         WHERE E2.dept_id = E.dept_id
         AND E2.employee_id != E.employee_id
     );

Q16. Find the Employee Name and their oldest project start date.
     SELECT E.employee_name, MIN(A.start_date) AS oldest_start_date
     FROM Employees E
     JOIN Assignments A ON E.employee_id = A.employee_id
     GROUP BY E.employee_name;

Q17. List all projects that were active in the year 2023 using DATE_FORMAT.
     SELECT project_name, DATE_FORMAT(start_date, '%Y-%m-%d') AS formatted_start
     FROM Projects
     WHERE DATE_FORMAT(start_date, '%Y') = '2023';

Q18. List all Employee Names. If an employee has no manager, display "CEO" using IFNULL.
     SELECT employee_name, IFNULL(CAST(manager_id AS CHAR), 'CEO') AS manager
     FROM Employees;

Q19. Find the top 3 Departments based on the total hours worked by all their employees.
     SELECT D.dept_name, SUM(A.hours_worked) AS total_hours
     FROM Departments D
     JOIN Employees E ON D.dept_id = E.dept_id
     JOIN Assignments A ON E.employee_id = A.employee_id
     GROUP BY D.dept_name
     ORDER BY total_hours DESC
     LIMIT 3;

Q20. Display the Project Name and a status using CASE ("High Activity" if total hours > 300, "Normal" otherwise).
     SELECT P.project_name,
            SUM(A.hours_worked) AS total_hours,
            CASE
                WHEN SUM(A.hours_worked) > 300 THEN 'High Activity'
                ELSE 'Normal'
            END AS status
     FROM Projects P
     JOIN Assignments A ON P.project_id = A.project_id
     GROUP BY P.project_name;

Q21. List the Employee Name and their Manager's Name, even if the employee has no manager
     or if a manager has no employees reporting to them.
     SELECT E.employee_name AS Employee, M.employee_name AS Manager
     FROM Employees E
     LEFT JOIN Employees M ON E.manager_id = M.employee_id;

Q22. Using a CTE, find the Project Name and the number of unique employees assigned to it.
     WITH ProjectEmployeeCount AS (
         SELECT project_id, COUNT(DISTINCT employee_id) AS unique_employees
         FROM Assignments
         GROUP BY project_id
     )
     SELECT P.project_name, PEC.unique_employees
     FROM Projects P
     JOIN ProjectEmployeeCount PEC ON P.project_id = PEC.project_id;

Q23. Find the Employee Name(s) who are assigned to the same projects as Employee ID 101.
     SELECT DISTINCT E.employee_name
     FROM Employees E
     JOIN Assignments A ON E.employee_id = A.employee_id
     WHERE A.project_id IN (
         SELECT project_id
         FROM Assignments
         WHERE employee_id = 101
     )
     AND E.employee_id != 101;
